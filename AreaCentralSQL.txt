CREATE DATABASE TABLER

CREATE TABLE TABLER.TBPRODUTOS (
  PROID INT PRIMARY KEY AUTO_INCREMENT,
  PRODESCRICAO VARCHAR(255),
  PROVALORUNI DECIMAL(10, 2),
  PROESTOQUE INT,
  PRODATAVENDA DATE DEFAULT NULL,
  PROTOTVENDA DECIMAL(10, 2) DEFAULT 0,
  PROEXCLUIDO BOOLEAN
);


CREATE TABLE TABLER.TBVENDAS (
  VENID INT PRIMARY KEY AUTO_INCREMENT,
  VENDATA DATE,
  PROID INT,
  VENQUANTIDADE INT,
  VENVALOR DECIMAL(10,2),
  VENVALORTOTAL DECIMAL(10,2),
  FOREIGN KEY (PROID) REFERENCES TABLER.TBPRODUTOS(PROID)
);
